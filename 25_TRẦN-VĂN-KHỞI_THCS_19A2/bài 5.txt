#bài 5
n =int(input("nhập n ="))

# S1
S1 =0
for i in range(1,n+1):
    S1 += i
print(f"vậy S1 sẽ có kết quả là {S1}")

#S2
S2 =1
for z in range(1,n):
    S2 *= z
print(f"vậy S2 sẽ có kết quả là {S2}")

#S3
S3 =0
for x in range(1,n+1):
    S3 += ((-1)**x)/x
print(f"vậy S3 sẽ có kết quả là {S3}")

#S4
S4 =0
for k in range(1,n+1):
    S4 += k/(k+1)
print(f"vậy S4 sẽ có kết quả là {S4}")
